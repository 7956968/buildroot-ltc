57 template <typename T>T parse(const std::string &str, bool &isOk)
58 {
59  std::istringstream iss(str);
60  T r = T();
61  iss >> std::noskipws >> r;
62  // noskipws considers leading white-space invalid
63  // Check the entire string was consumed and if either failbit or
64  // badbit is set to know if conversion was correct
65  isOk = iss.eof() && !iss.fail();
66  return r;
67 }
