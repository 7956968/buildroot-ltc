111  const char* file, uint32_t line, int verif)
112 {
113  int ret = 0;
114  if (irqs_disabled())
115  {
116  printk(KERN_WARNING "WARNING: %s with irq disabled! %s:%u",
117  fct, file, line);
118  if (IMG_KMALLOC_FLAG != GFP_ATOMIC)
119  {
120  dump_stack();
121  ret++; // so it only prints the warning and does not allocate
122  }
123  }
124  if (0 == size)
125  {
126  printk(KERN_ERR \
127  "ERROR: %s (%s:%u) tries to allocate a 0 sized buffer!",
128  fct, file, line);
129  }
130  if (verif && size > 2 * PAGE_SIZE)
131  {
132  /* kernel buddy allocator (on kernels >2.6.35) does not suffer
